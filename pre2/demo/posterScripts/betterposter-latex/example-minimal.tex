\begin{document}

  }{

      \textbf{Take a picture} to download the full paper
    }
  }
}{

  \title{Your Poster Title}
  \author{Author One \and Author Two}
  \institution{Your Institution}

  \section{Introduction}
  \begin{itemize}
    \item Point one.
    \item Point two.
  \end{itemize}

  \section{Methods}
  Brief methods here.

  \section{Results}
  Key results or one figure.

  \vfill
  \includegraphics[width=\textwidth]{img/logo}
}{
  %%% RIGHT COLUMN (supplementary)
  Extra figures or material.

  \includegraphics[width=\textwidth]{img/supplementary}
}
\end{document}
